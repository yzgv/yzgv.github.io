工资计算器

//作者：04F2104 袁珍贵
var bs, as, ts;
document.getElementById("calc").onclick = function() {
	bs = Number(document.getElementById("input").value);
	output = document.getElementById("output");
	if(bs >= 1000) {
		as = bs * 0.2;
	} else if(bs >= 800) {
		as = bs * 0.15;
	} else if(bs >= 0) {
		as = bs * 0.1;
	} else {
		output.innerHTML = "";
		var msg = document.createTextNode("输入错误");
		output.appendChild(msg)
		return;
	}
	ts = bs + as;
	output.innerHTML = "";
	var div = document.createElement("div");
	div.innerHTML = "基本工资：" + bs + "元<br />增加工资：" + as + "元<br />总工资：" + ts + "元";
	output.appendChild(div);
}
